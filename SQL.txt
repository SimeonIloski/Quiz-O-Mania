--Tabela za prasanjata
create table Questions(
	QuestionID int primary key,
	Question varchar(60),
	Picture varchar(60),
	Answer varchar(60),
	Hint varchar(60),
	Category varchar(60)
);

--Tabela za Users
create table Users(
	UserName varchar(60),
	Score int,
	Mode varchar(20)
);

--Tabela za Active User
create table ActiveUser(
	UserName varchar(60),
	GameMode varchar(20),
	Category varchar(60)
);
